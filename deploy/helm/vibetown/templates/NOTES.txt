{{- else }}

To access the application, set up a port-forward:

  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "vibetown.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME {{ .Values.server.httpPort }}:{{ .Values.server.httpPort }}

Then open http://localhost:{{ .Values.server.httpPort }} in your browser.

{{- end }}

To check the status of the deployment:

  kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/instance={{ .Release.Name }}"
